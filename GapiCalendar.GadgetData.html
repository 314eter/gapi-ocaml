A gadget that extends this event.

type t = {
   _type : string; (*The gadget's type.*)
   title : string; (*The gadget's title.*)
   link : string; (*The gadget's URL.*)
   iconLink : string; (*The gadget's icon URL.*)
   width : int; (*The gadget's width in pixels. Optional.*)
   height : int; (*The gadget's height in pixels. Optional.*)
   display : string; (*The gadget's display mode. Optional. Possible values are:
  • "icon" - The gadget displays next to the event's title in the calendar view.
  • "chip" - The gadget displays when the event is clicked.
*)

}
